# Функция для вычисления факториала
def factorial(n):
    # Проверяем, является ли n отрицательным
    if n < 0:
        return "Факториал не определен для отрицательных чисел."
    # Факториал нуля равен 1
    elif n == 0:
        return 1
    else:
        result = 1
        # В цикле умножаем числа от 1 до n
        for i in range(1, n + 1):
            result *= i
        return result

# Запрос числа у пользователя
num = int(input("Введите число для вычисления факториала: "))

# Вызов функции и вывод результата
print(f"Факториал числа {num} равен: {factorial(num)}")
